Core Message Interface import crypto, { randomUUID } from "crypto"; import { firestore } from "../server/services/database.js"; import { createDecompositionPlan } from "./createDecompositionPlan.js"; import { expandWorkerSteps } from "./expandWorkerSteps.js"; import { aggregateArtifacts } from "./aggregateArtifacts.js"; import type { ExecutionEnvelope } from "../types/executionEnvelope.js"; export type USMessageType = | "#us#.task.plan" | "#us#.task.assign" | "#us#.artifact.produce" | "#us#.artifact.reference" | "#us#.evaluation.score" | "#us#.execution.complete"; export interface USMessage { protocol: "#us#"; version: "1.0"; message_type: USMessageType; execution: { envelope_id: string; step_id: string; }; identity: { agent_id: string; identity_fingerprint: string; }; authority: { lease_id?: string; }; payload: { role?: "COO" | "Researcher" | "Worker" | "Grader"; artifact_type?: string; content?: string; artifact_id?: string; artifact_ids?: string[]; work_unit?: Record<string, unknown> | null; aggregated_content?: string; status?: string; score?: number; summary?: string; [key: string]: unknown; }; metadata: { timestamp: string; }; } export interface ExecutionMessage { message_id: string; envelope_id: string; step_id: string; message_type: string; agent_id: string; identity_fingerprint: string; payload: Record<string, unknown>; created_at: string; } 4. Message Factory (STRICT) No one should manually construct messages. export function createUSMessage( input: Omit<USMessage, "protocol" | "version" | "metadata"> ): USMessage { return { protocol: "#us#", version: "1.0", ...input, metadata: { timestamp: new Date().toISOString() } }; } 5. Message Persistence (MANDATORY) Collection: execution_messages/{message_id} export async function storeUSMessage(msg: USMessage) { const message_id = randomUUID(); await firestore.collection("execution_messages").doc(message_id).set({ message_id, envelope_id: msg.execution.envelope_id, step_id: msg.execution.step_id, message_type: msg.message_type, agent_id: msg.identity.agent_id, identity_fingerprint: msg.identity.identity_fingerprint, payload: msg.payload, created_at: new Date().toISOString() }); await firestore.collection("execution_traces").add({ envelope_id: msg.execution.envelope_id, message_id, agent_id: msg.identity.agent_id, identity_fingerprint: msg.identity.identity_fingerprint, event_type: msg.message_type, timestamp: new Date().toISOString() }); return message_id; } 6. Message Router This decides what happens next. export async function handleUSMessage(msg: USMessage): Promise<USMessage | null> { switch (msg.message_type) { case "#us#.task.plan": return handleTaskPlan(msg); case "#us#.task.assign": return handleTaskAssign(msg); case "#us#.artifact.produce": return handleArtifactProduce(msg); case "#us#.evaluation.score": return handleEvaluation(msg); case "#us#.execution.complete": return handleExecutionComplete(msg); default: throw new Error("UNKNOWN_MESSAGE_TYPE"); } } 7. Handlers (Core Logic) 7.1 Task Plan COO step. This is where planning semantics live, but the envelope already holds the workflow graph. So this handler mainly records that planning executed successfully. async function handleTaskPlan(msg: USMessage): Promise<USMessage | null> { return createUSMessage({ message_type: "#us#.task.assign", execution: msg.execution, identity: msg.identity, authority: msg.authority, payload: { role: "Researcher" } }); } 7.2 Task Assign Researcher step. This is where decomposition happens. The Researcher: decides how many worker units to create creates a decomposition plan expands Worker steps into the envelope defines aggregation strategy for the Grader async function handleTaskAssign(msg: USMessage): Promise<USMessage | null> { const envelopeSnap = await firestore .collection("execution_envelopes") .doc(msg.execution.envelope_id) .get(); if (!envelopeSnap.exists) { throw new Error("ENVELOPE_NOT_FOUND"); } const envelope = envelopeSnap.data() as ExecutionEnvelope; const workerAgentIds = Object.keys(envelope.identity_contexts).filter((agentId) => envelope.steps.some( (s) => s.role === "Worker" && s.assigned_agent_id === agentId ) ); const decompositionPlan = createDecompositionPlan({ objective: envelope.root_task_id, worker_agent_ids: workerAgentIds.length ? workerAgentIds : [msg.identity.agent_id], parent_step_id: msg.execution.step_id }); await expandWorkerSteps({ envelope_id: msg.execution.envelope_id, decomposition_plan: decompositionPlan }); return null; } 7.3 Artifact Produce Worker step. Each Worker produces one artifact. async function handleArtifactProduce(msg: USMessage): Promise<USMessage | null> { const content = buildWorkerArtifactContent(msg); const artifact_type = String(msg.payload.artifact_type || "report"); const artifact_id = crypto.randomUUID(); await firestore.collection("artifacts").doc(artifact_id).set({ artifact_id, execution_id: msg.execution.envelope_id, produced_by_agent: msg.identity.agent_id, identity_fingerprint: msg.identity.identity_fingerprint, artifact_type, artifact_content: content, created_at: new Date().toISOString() }); await attachArtifactToStep(msg.execution.envelope_id, msg.execution.step_id, artifact_id); await attachArtifactToEnvelope(msg.execution.envelope_id, artifact_id); return createUSMessage({ message_type: "#us#.artifact.reference", execution: msg.execution, identity: msg.identity, authority: msg.authority, payload: { role: "Worker", artifact_id } }); } 7.4 Evaluation Grader step. This is where fan-in happens. The Grader: collects all completed Worker artifacts aggregates them according to the stored decomposition plan evaluates the aggregated result emits execution completion async function handleEvaluation(msg: USMessage): Promise<USMessage | null> { const envelopeSnap = await firestore .collection("execution_envelopes") .doc(msg.execution.envelope_id) .get(); if (!envelopeSnap.exists) { throw new Error("ENVELOPE_NOT_FOUND"); } const envelope = envelopeSnap.data() as ExecutionEnvelope; const workerSteps = (envelope.steps || []).filter( (s) => s.role === "Worker" && s.step_type === "produce_artifact" && s.status === "completed" && s.output_ref?.artifact_id ); const artifactIds = workerSteps .map((s) => s.output_ref.artifact_id) .filter(Boolean) as string[]; const aggregatedContent = await aggregateArtifacts(artifactIds); return createUSMessage({ message_type: "#us#.execution.complete", execution: msg.execution, identity: msg.identity, authority: msg.authority, payload: { status: "success", role: "Grader", score: 0.95, summary: "Aggregated output validated", artifact_ids: artifactIds, aggregated_content: aggregatedContent } }); } 7.5 Execution Complete Final step. Stores the final output and marks the envelope complete. async function handleExecutionComplete(msg: USMessage): Promise<USMessage | null> { const final_artifact_id = crypto.randomUUID(); await firestore.collection("artifacts").doc(final_artifact_id).set({ artifact_id: final_artifact_id, execution_id: msg.execution.envelope_id, produced_by_agent: msg.identity.agent_id, identity_fingerprint: msg.identity.identity_fingerprint, artifact_type: "final_result", artifact_content: msg.payload.aggregated_content || "", created_at: new Date().toISOString() }); await attachArtifactToEnvelope(msg.execution.envelope_id, final_artifact_id); await firestore .collection("execution_envelopes") .doc(msg.execution.envelope_id) .update({ status: "completed", updated_at: new Date().toISOString() }); return null; } 8. Helper Functions These keep the core handlers clean. function buildWorkerArtifactContent(msg: USMessage): string { const workUnit = msg.payload.work_unit as Record<string, unknown> | null | undefined; if (workUnit) { const title = String(workUnit.title || "Untitled Work Unit"); const objective = String(workUnit.objective || ""); const instructions = String(workUnit.instructions || ""); return [ `# ${title}`, "", `Objective: ${objective}`, "", `Instructions: ${instructions}`, "", `Generated by agent ${msg.identity.agent_id}` ].join("\n"); } return `Generated artifact by ${msg.identity.agent_id}`; } async function attachArtifactToStep( envelopeId: string, stepId: string, artifactId: string ) { const ref = firestore.collection("execution_envelopes").doc(envelopeId); await firestore.runTransaction(async (tx) => { const snap = await tx.get(ref); if (!snap.exists) throw new Error("ENVELOPE_NOT_FOUND"); const envelope = snap.data() as ExecutionEnvelope; const steps = (envelope.steps || []).map((step) => { if (step.step_id !== stepId) return step; return { ...step, output_ref: { ...step.output_ref, artifact_id: artifactId }, updated_at: new Date().toISOString() }; }); tx.update(ref, { steps, updated_at: new Date().toISOString() }); }); } async function attachArtifactToEnvelope(envelopeId: string, artifactId: string) { const ref = firestore.collection("execution_envelopes").doc(envelopeId); await firestore.runTransaction(async (tx) => { const snap = await tx.get(ref); if (!snap.exists) throw new Error("ENVELOPE_NOT_FOUND"); const envelope = snap.data() as ExecutionEnvelope; const artifact_refs = [...(envelope.artifact_refs || []), artifactId]; tx.update(ref, { artifact_refs, updated_at: new Date().toISOString() }); }); } 9.
